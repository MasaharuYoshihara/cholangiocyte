import numpy as np
from scipy import optimize
def func(x):
    return[100-x[0]-11*x[0]*x[1],
          10/(1+x[2])-x[1]-11*x[0]*x[1],
          1000000*(x[0]*x[1])**3/(300000+(x[0]*x[1])**3)-x[2]]
sol = optimize.root(func,[0,0,0],method="lm")
